int compare = 0;

void swap(int *x, int *y) {
	int temp = *x;
	*x = *y;
	*y = temp;
}

void insertion_sort(int list[], int n) {
	int i, j, key;
	compare = 0;
	for (i = 1; i < n; i++) {
		key = list[i];
		for (j = i - 1; j >= 0 && list[j] > key; j--) {
			compare += 1;
			list[j + 1] = list[j];
		}
		list[j + 1] = key;
	}
}

void bubble_sort(int list[], int n) {
	int i, j = 0, flag = true;
	compare = 0;
	while ((j < n - 1) && flag) {
		flag = false;
		for (i = n - 1; i > j; i--) {
			compare += 1;
			if (list[i] < list[i - 1]) {
				flag = true;
				swap(&list[i], &list[i - 1]);
			}
		}
		j++;
	}
}
